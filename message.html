Below is an example of a Python script that sends an HTML email:


import smtplib
from email.message import EmailMessage

email_subject = "Email test Pythonist"

sender_email_address = "example@gmail.com"
receiver_email_address = "receiver@gmail.com"
email_password = "APP_PASSWORD"

email_smtp = "smtp.gmail.com"
message = EmailMessage()

message.set_content("Hello from Python!")

message['Subject'] = email_subject
message['From'] = sender_email_address
message['To'] = receiver_email_address

server = smtplib.SMTP(email_smtp, 587)
server.starttls()
server.login(sender_email_address, email_password)
server.send_message(message)
server.quit()
